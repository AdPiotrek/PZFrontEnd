import { GroupComponent } from './logged-in/content/group-managment/group/group.component';

import { UploadFileToGroupComponent } from './logged-in/content/upload-file-to-group/upload-file-to-group.component';

  declarations: [

    BrowserModule,

    MatTableModule,
